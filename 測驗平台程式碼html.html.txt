        window.switchView = function(viewName) {
            state.view = viewName;
            const contentArea = document.getElementById('content-area');
            const quizNav = document.getElementById('quiz-navigation');
            const statusDiv = document.getElementById('quiz-status');

            contentArea.innerHTML = '';
            quizNav.classList.add('hidden');
            statusDiv.classList.add('hidden');

            if (viewName === 'filter') {
                renderFilterScreen();
            } else if (viewName === 'quiz') {
                quizNav.classList.remove('hidden');
                statusDiv.classList.remove('hidden');
                renderQuizScreen();
                updateQuizStatus();
            } else if (viewName === 'result') {
                renderResultScreen();
            }
        }

                        </h2>
                        <div class="grid grid-cols-2 gap-4">
                            ${optionsHtml}
                        </div>
                    </div>
                `;
            }

            // 提取所有章節和領域選項
            const chapterOptions = Array.from(new Set(questionBank.map(q => q.chapter))).sort((a, b) => {
                const numA = parseInt(a.replace('CH', ''));
                const numB = parseInt(b.replace('CH', ''));
                return numA - numB;
            });
            
            const topicOptions = Array.from(new Set(questionBank.map(q => q.topic))).sort();
            // 隨機抽題數量選項 (已移除 30 和 50 題)
            const countOptions = [10, 20]; 

            // 確保初始狀態填充預設值
            if (state.filters.chapters.length === 0) state.filters.chapters = [...chapterOptions];
            if (state.filters.topics.length === 0) state.filters.topics = [...topicOptions];
            if (!state.filters.count || !countOptions.includes(state.filters.count)) state.filters.count = countOptions[0];


            contentArea.innerHTML = `
                <div class="max-w-3xl mx-auto p-8 bg-white rounded-2xl shadow-2xl border-t-4 border-accent-teal">
                    <h1 class="text-3xl font-extrabold text-center text-primary-indigo mb-8">
                        設定您的專屬刷題清單
                    </h1>

                    <!-- 1. 章節篩選 (可複選) -->
                    ${renderCheckboxGroup('章節範圍', chapterOptions, 'chapters')}

                    <!-- 2. 題目領域篩選 (可複選) -->
                    ${renderCheckboxGroup('題目領域', topicOptions, 'topics')}

                    <!-- 3. 隨機抽題數量 (單選) -->
                    ${renderRadioGroup('隨機抽題數量', countOptions, 'count')}

                    <!-- 開始測驗按鈕 -->
                    <div class="text-center mt-12">
                        <button onclick="startQuiz()" class="w-full sm:w-2/3 bg-primary-indigo text-white py-4 px-8 rounded-full shadow-2xl hover:bg-primary-hover transition duration-300 transform hover:scale-[1.05] font-bold text-xl ring-4 ring-indigo-200 hover:ring-indigo-300">
                            <i data-lucide="rocket" class="inline-block w-6 h-6 mr-2"></i>
                            準備好了，開始訓練！
                        </button>
                    </div>
                </div>
            `;
            lucide.createIcons(); // 重新渲染 Lucide 圖標
        }

        // 全域函數供 HTML 呼叫
        window.handleFilterChange = (element, type) => {
            const value = element.value;

            if (type === 'count') {
                // 單選 (Radio)
                state.filters[type] = parseInt(value, 10);
            } else {
                // 複選 (Checkbox)
                if (element.checked) {
                    if (!state.filters[type].includes(value)) {
                        state.filters[type].push(value);
                    }
                } else {
                    state.filters[type] = state.filters[type].filter(v => v !== value);
                }
            }
            // 重新渲染篩選器以更新視覺狀態
            renderFilterScreen();
        };

        window.startQuiz = () => {
            const { chapters, topics, count } = state.filters;

            // 1. 過濾題目
            let pool = questionBank.filter(q =>
                chapters.includes(q.chapter) && topics.includes(q.topic)
            );

            if (pool.length === 0) {
                // 使用自定義訊息框代替 alert()
                showCustomMessage('錯誤', '根據您的篩選條件，題庫中找不到任何題目。請調整您的選擇。');
                return;
            }

            // 2. 隨機抽題 (Shallow copy and Fisher-Yates shuffle)
            const shuffled = [...pool].sort(() => 0.5 - Math.random());
            
            // 3. 限制數量
            state.filteredQuestions = shuffled.slice(0, count);
            
            // 4. 初始化測驗狀態
            state.selectedAnswers = new Array(state.filteredQuestions.length).fill(null);
            state.currentQuestionIndex = 0;

            // 5. 切換到測驗畫面
            switchView('quiz');
        };

        // ====================================================================
        // 2. 測驗頁面 (QuizScreen) 渲染與邏輯
        // ====================================================================

        function updateQuizStatus() {
            const statusDiv = document.getElementById('quiz-status');
            const total = state.filteredQuestions.length;
            const current = state.currentQuestionIndex + 1;
            const answeredCount = state.selectedAnswers.filter(a => a !== null).length;

            statusDiv.innerHTML = `
                <span class="text-lg font-bold mr-4 text-accent-teal">Q${current}/${total}</span>
                <span class="text-sm opacity-90">已作答: ${answeredCount} 題</span>
            `;

            // 更新導航按鈕狀態
            document.getElementById('prev-btn').disabled = current === 1;
            document.getElementById('next-btn').disabled = current === total;

            // 顯示完成按鈕
            if (answeredCount === total) {
                document.getElementById('finish-btn').classList.remove('hidden');
            } else {
                document.getElementById('finish-btn').classList.add('hidden');
            }
            
            renderJumpButtons();
        }

        function renderJumpButtons() {
            const jumpContainer = document.getElementById('question-jump');
            const total = state.filteredQuestions.length;
            
            jumpContainer.innerHTML = Array.from({ length: total }, (_, i) => {
                const isCurrent = i === state.currentQuestionIndex;
                const isAnswered = state.selectedAnswers[i] !== null;

                let classes = 'w-7 h-7 flex items-center justify-center rounded-full text-xs font-bold transition duration-150 transform hover:scale-110';
                
                if (isCurrent) {
                    classes += ' bg-primary-indigo text-white shadow-lg ring-2 ring-indigo-400';
                } else if (isAnswered) {
                    classes += ' bg-accent-teal/80 text-white hover:bg-accent-teal border border-teal-400';
                } else {
                    classes += ' bg-gray-200 text-gray-700 hover:bg-gray-300 border border-gray-300';
                }

                // 修正：確保 jumpToQuestion 函數被正確調用
                return `<button onclick="jumpToQuestion(${i})" class="${classes}" title="${isAnswered ? '已作答' : '未作答'}">${i + 1}</button>`;
            }).join('');
        }

        function renderQuizScreen() {
            const contentArea = document.getElementById('content-area');
            const index = state.currentQuestionIndex;
            const questionData = state.filteredQuestions[index];
            const currentAnswer = state.selectedAnswers[index];
            
            if (!questionData) return;

            const options = ['A', 'B', 'C', 'D'];

            const optionsHtml = options.map(opt => `
                <button 
                    onclick="handleAnswer('${opt}')" 
                    class="option-btn text-left p-4 mb-4 border-2 rounded-2xl transition duration-200 ease-in-out shadow-lg 
                           ${currentAnswer === opt ? 'bg-indigo-100 border-primary-indigo ring-4 ring-indigo-300 shadow-2xl' : 'bg-white border-gray-200 hover:bg-indigo-50 hover:border-primary-hover hover:shadow-xl'}">
                    <span class="font-extrabold text-primary-indigo mr-3 text-lg">${opt}.</span>
                    <span class="text-text-primary text-base">${questionData[opt]}</span>
                </button>
            `).join('');

            contentArea.innerHTML = `
                <div class="max-w-3xl mx-auto">
                    <div class="text-sm font-semibold text-gray-500 mb-4 flex items-center justify-between">
                        <span><i data-lucide="book-open" class="inline-block w-4 h-4 mr-1"></i> 章節: ${questionData.chapter}</span>
                        <span><i data-lucide="trending-up" class="inline-block w-4 h-4 mr-1"></i> 領域: ${questionData.topic}</span>
                    </div>
                    
                    <!-- 題目主體 -->
                    <div class="bg-white p-8 rounded-2xl shadow-xl border-l-4 border-accent-teal mb-8">
                        <h2 class="text-2xl font-extrabold text-text-primary leading-relaxed">
                            Q${index + 1}: ${questionData.question}
                        </h2>
                    </div>

                    <!-- 選項區 -->
                    <div id="options-container" class="flex flex-col">
                        ${optionsHtml}
                    </div>

                    <div class="mt-8 text-center">
                        <p id="feedback-msg" class="text-lg font-bold text-indigo-500 p-2 bg-indigo-50 rounded-full inline-block shadow-inner">
                            ${currentAnswer ? '✅ 您已選擇答案，請點擊下一題或完成按鈕。' : '請選擇您的答案，讓系統記錄您的作答。'}
                        </p>
                    </div>
                </div>
            `;
            lucide.createIcons(); // 重新渲染 Lucide 圖標
        }

        // ====================================================================
        // 3. 結果頁面 (ResultScreen) 渲染與邏輯
        // ====================================================================

        function calculateScore() {
            let correctCount = 0;
            const total = state.filteredQuestions.length;
            
            state.filteredQuestions.forEach((q, i) => {
                if (state.selectedAnswers[i] === q.answer) {
                    correctCount++;
                }
            });

            const score = Math.round((correctCount / total) * 100);
            return { correctCount, total, score };
        }

        function renderResultScreen() {
            const contentArea = document.getElementById('content-area');
            const { correctCount, total, score } = calculateScore();

            // 根據分數給予不同鼓勵
            const encouragement = score >= 80 ? 
                "🎉 表現優異！您已掌握大部分重點！" : 
                (score >= 60 ? "👍 幹得好！繼續努力，目標滿分！" : "💪 繼續加油！多練習錯題，會越學越好！");

            // 生成結果詳情 HTML
            const resultsDetailHtml = state.filteredQuestions.map((q, i) => {
                const userAnswer = state.selectedAnswers[i];
                const isCorrect = userAnswer === q.answer;
                const icon = isCorrect 
                    ? `<i data-lucide="award" class="w-6 h-6 text-accent-teal"></i>` 
                    : `<i data-lucide="alert-triangle" class="w-6 h-6 text-rose-600"></i>`;
                
                // 獲取使用者選擇的選項內容，如果未作答則顯示 N/A
                const userAnswerText = userAnswer ? q[userAnswer] : '未作答 (N/A)';

                return `
                    <div id="q-detail-${i}" class="p-5 border-b border-gray-100 transition duration-150 rounded-lg my-2 ${isCorrect ? 'bg-teal-50 shadow-sm' : 'bg-rose-50 shadow-sm'}">
                        <div class="flex items-start space-x-3 mb-3">
                            ${icon}
                            <div class="flex-1">
                                <h3 class="text-lg font-bold text-text-primary">Q${i + 1}: ${q.question}</h3>
                            </div>
                            <span class="text-sm font-bold ml-auto p-1.5 rounded-full px-3 shadow-md ${isCorrect ? 'bg-accent-teal text-white' : 'bg-rose-500 text-white'}">${isCorrect ? '✅ 正確' : '❌ 錯誤'}</span>
                        </div>
                        
                        <div class="ml-9 text-sm space-y-2">
                            <p class="font-semibold">您的答案: <span class="${isCorrect ? 'text-accent-teal font-extrabold' : 'text-rose-700 font-extrabold'}">${userAnswer || 'N/A'}</span>. ${userAnswerText}</p>
                            <p class="font-semibold">正確答案: <span class="text-indigo-700 font-extrabold">${q.answer}</span>. ${q[q.answer]}</p>
                            
                            <!-- 基礎解析 -->
                            <div class="mt-4 p-4 bg-white rounded-xl border border-indigo-200 shadow-inner">
                                <strong class="text-primary-indigo flex items-center"><i data-lucide="message-square" class="w-4 h-4 mr-2"></i> 詳解 (基礎):</strong> 
                                <p class="mt-1 text-gray-700">${q.explanation}</p>
                            </div>
                        </div>
                    </div>
                `;
            }).join('');

            contentArea.innerHTML = `
                <div class="max-w-4xl mx-auto">
                    <!-- 分數總結區 (更動態的漸變色和陰影) -->
                    <div class="text-center bg-gradient-to-br from-indigo-500 to-purple-500 text-white p-10 rounded-3xl shadow-2xl mb-10 transform transition duration-500 hover:scale-[1.01] ring-8 ring-indigo-300/50">
                        <h2 class="text-3xl font-light mb-4">${encouragement}</h2>
                        <div class="text-7xl font-extrabold leading-none mb-4 tracking-tight">
                            ${score}<span class="text-4xl align-top">%</span>
                        </div>
                        <div class="flex justify-center items-center space-x-8 mt-4 text-xl font-medium">
                            <p><i data-lucide="hash" class="inline-block w-5 h-5 mr-1 text-accent-teal"></i> 總題數: ${total}</p>
                            <p><i data-lucide="check" class="inline-block w-5 h-5 mr-1 text-accent-teal"></i> 答對: ${correctCount}</p>
                            <p><i data-lucide="x" class="inline-block w-5 h-5 mr-1 text-accent-teal"></i> 答錯: ${total - correctCount}</p>
                        </div>
                    </div>

                    <!-- 詳解與答案區 -->
                    <div class="bg-white rounded-2xl shadow-xl overflow-hidden">
                        <h2 class="text-2xl font-bold p-5 bg-indigo-100 text-primary-indigo border-b border-indigo-200 flex items-center">
                            <i data-lucide="list-checks" class="w-6 h-6 mr-3"></i>
                            詳解與錯誤複習清單
                        </h2>
                        <div class="divide-y divide-gray-100 p-2">
                            ${resultsDetailHtml}
                        </div>
                    </div>

                    <!-- 返回按鈕 -->
                    <div class="text-center mt-10 pb-4">
                        <button onclick="switchView('filter')" class="bg-gray-700 text-white py-3 px-8 rounded-full shadow-lg hover:bg-gray-800 transition duration-300 transform hover:scale-[1.02] font-bold text-lg">
                            <i data-lucide="repeat-2" class="inline-block w-5 h-5 mr-2"></i>
                            重新設定訓練
                        </button>
                    </div>
                </div>
            `;
            lucide.createIcons(); // 重新渲染 Lucide 圖標
        }

        // ====================================================================
        // 導航和控制
        // ====================================================================

        window.handleAnswer = (option) => {
            state.selectedAnswers[state.currentQuestionIndex] = option;
            renderQuizScreen(); // 重新渲染以更新選項的視覺狀態和作答狀態
            updateQuizStatus(); // 確保作答狀態和完成按鈕狀態同步更新
        };

        /**
         * 跳轉到特定題號
         * @param {number} index - 題目的索引 (0-based)
         */
        window.jumpToQuestion = (index) => {
            if (index >= 0 && index < state.filteredQuestions.length) {
                state.currentQuestionIndex = index;
                renderQuizScreen();
                // 修正: 確保跳轉後重新渲染底部跳轉按鈕和頂部狀態
                updateQuizStatus(); 
            }
        };

        /**
         * 改變當前題目索引
         * @param {number} delta - 變化量 (-1 for previous, 1 for next)
         */
        window.changeQuestion = (delta) => {
            const newIndex = state.currentQuestionIndex + delta;
            jumpToQuestion(newIndex);
        };

        window.showResults = () => {
            // 確認是否所有題目都已作答 (以防萬一)
            const unanswered = state.selectedAnswers.filter(a => a === null).length;
            if (unanswered > 0) {
                // 使用自定義訊息框代替 alert()
                showCustomMessage('確認完成', `您還有 ${unanswered} 題尚未作答。確定要提前結束並查看結果嗎？`, () => {
                    switchView('result');
                });
            } else {
                switchView('result');
            }
        };

        // ====================================================================
        // Custom Message Box (替代 alert/confirm)
        // ====================================================================

        function showCustomMessage(title, message, onConfirm = null) {
            const existingModal = document.getElementById('custom-modal');
            if (existingModal) existingModal.remove();

            const isConfirm = onConfirm !== null;

            const modalHtml = `
                <div id="custom-modal" class="fixed inset-0 bg-black bg-opacity-60 flex items-center justify-center z-50 p-4 backdrop-blur-sm">
                    <div class="bg-white rounded-2xl shadow-2xl w-full max-w-sm p-6 transform transition-all duration-300 scale-100 border-t-8 border-primary-indigo">
                        <h3 class="text-xl font-bold mb-3 text-primary-indigo border-b pb-2 flex items-center">
                            <i data-lucide="bell" class="w-5 h-5 mr-2"></i>
                            ${title}
                        </h3>
                        <p class="text-text-primary mb-6">${message}</p>
                        <div class="flex justify-end space-x-3">
                            ${isConfirm ? `
                                <button id="modal-cancel-btn" class="px-4 py-2 bg-gray-200 text-gray-700 rounded-full hover:bg-gray-300 transition font-medium">取消</button>
                                <button id="modal-confirm-btn" class="px-4 py-2 bg-primary-indigo text-white rounded-full hover:bg-primary-hover transition font-medium shadow-md">確定</button>
                            ` : `
                                <button id="modal-close-btn" class="px-4 py-2 bg-primary-indigo text-white rounded-full hover:bg-primary-hover transition font-medium shadow-md">關閉</button>
                            `}
                        </div>
                    </div>
                </div>
            `;
            document.body.insertAdjacentHTML('beforeend', modalHtml);
            lucide.createIcons();

            const modal = document.getElementById('custom-modal');
            const close = () => modal.remove();

            if (isConfirm) {
                document.getElementById('modal-cancel-btn').onclick = close;
                document.getElementById('modal-confirm-btn').onclick = () => {
                    close();
                    onConfirm();
                };
            } else {
                document.getElementById('modal-close-btn').onclick = close;
            }
        }
        window.showCustomMessage = showCustomMessage;

        // 應用程式啟動點
        window.onload = function() {
            switchView('filter');
        };

    </script>
</body>
